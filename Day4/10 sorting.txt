void sort (int a[],int l)
{
int i,j,t;
for(i=0;i<l;i++)
{
	for (j=i+1;j<=l;j++)
	{
		if(a[i]<a[j])
		{
		t=a[i];
		a[i]=a[j];
		a[j]=t;
		}
	}
	
}
