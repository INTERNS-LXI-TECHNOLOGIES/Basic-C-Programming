#include <stdio.h>
int Armstrong(int num)
{
    int qsum=0,r,n;
    n=num;
    while(n>0)
    {
        r=n%10;
        qsum=qsum+(r*r*r);
        n=n/10;
    }
    if(qsum==num)
        return 1;
    else
        return 0;
}
int main()
{
    int num,result;
    printf("Enter the number:");
    scanf("%d",&num);
    result = Armstrong(num);
    if(result==1)
        printf("%d is armstrong number",num);
    else
        printf("%d is not an armstrong number",num);
    return 0;
}
