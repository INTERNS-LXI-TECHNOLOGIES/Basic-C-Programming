void strong(int num2)
{
    int i,x,sum,rem,fact;
    for(num2=100;num2<=1000;num2++)
    {
        x=num2;
        sum=0;
        while(x!=0)
        {
            rem=x%10;
            fact=1;
            for(i=rem;i>=1;i--)
            {
                fact=fact*i;
                
            }
            sum=sum+fact;
            x=x/10;
        }
    if(sum==num2)
    printf("strong series from 100 and 1000 is :%d \t",num2);
    }
}
