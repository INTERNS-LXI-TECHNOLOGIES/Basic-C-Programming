#include <stdio.h>
#include <stdlib.h>



void main()
{
	int i,r,b,x;
	printf("Enter no.of rows:\n");
	scanf("%d",&r);
	
	for(i=1;i<=r;i++)
	{	
		
		for(b=i;b<r;b++)
		{
			printf(" ");
		}
		for(x=1;x<=i;x++)
		{
			printf(" *");
		}
		printf("\n");
	}
	for(i=r;i>=1;i--)
	{	
		
		for(b=r;b>i;b--)
		{
			printf(" ");
		}
		for(x=1;x<=i;x++)
		{
			printf(" *");
		}
		printf("\n");
	}
	
}
