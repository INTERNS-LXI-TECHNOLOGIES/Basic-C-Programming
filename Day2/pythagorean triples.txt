#include<stdio.h>
int main()
{
    int a, b,c,limit=250;
   
  printf("pythagorean triples:");
 
    for(a=1;a<=limit;a++)
    {
        for(b=a+1;b<=limit;b++)
        {
 	for(c=b+1;b<=limit;c++)
	{
          if(a*a + b*b == c*c) 
            {
                printf(" %d, %d, %d \n",a,b,c);
		}
            }
	}
    }
return 0;
}
