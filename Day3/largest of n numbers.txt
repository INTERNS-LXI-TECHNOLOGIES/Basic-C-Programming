#include<stdio.h>  
  
int main()  
{  
    int limit, num, largest = 0;  
  
    printf("Enter the limit\n");  
    scanf("%d", &limit);  
  
    printf("Enter %d numbers\n", limit);  
  
    scanf("%d", &num);  
    largest = num;  
    limit = limit - 1;  
  
    while(limit > 0)  
    {  
        if(num > largest)  
        {  
            largest = num;  
        }  
        limit--;  
    }  
  
    printf("largest number is %d\n", largest);  
  
    return 0;  
}
